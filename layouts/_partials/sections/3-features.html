See it in action

Compare how Diesel transforms complex SQL into readable, safe Rust code.

Rust (Diesel)

src/models.rs

use diesel::prelude::*;

#[derive(Queryable, Selectable)]

#[diesel(check_for_backend(diesel::pg::Pg))]

pub struct Post {

pub id: i32,

pub title: String,

pub body: String,

pub published: bool,

}

Equivalent SQL

SELECT * FROM posts

WHERE published = 1

ORDER BY id DESC;
